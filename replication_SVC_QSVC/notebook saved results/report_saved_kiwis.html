def find_top_index(results, n, metric=4, show_result=False):
    sorted_results = sorted(results, key=lambda x: x[metric], reverse=True)
    
    if show_result:
        for sr in sorted_results:
            print(sr)
    top_index = [r[0] for r in sorted_results[:n]]
    return top_index
